148 void ShapeWorksGroom::center(int which) {
149  if (this->verbose_) {
150  std::cout << "*** RUNNING TOOL: center on " <<
151  (which == -1 ? "all" : std::to_string(which)) << std::endl;
152  }
153  auto start = (which == -1 ? 0 : which);
154  auto end = (which == -1 ? this->images_.size() : which + 1);
155  for (size_t i = start; i < end; i++) {
156  auto img = this->images_[i];
157  ImageType::PointType origin = img->GetOrigin();
158 
159  // Copy the original image and find the center of mass.
160  ImageType::Pointer simg = ImageType::New();
161  simg->CopyInformation(img);
162  simg->SetRegions(img->GetLargestPossibleRegion());
163  simg->Allocate();
164 
165  itk::ImageRegionIteratorWithIndex<ImageType> oit(img, img->GetLargestPossibleRegion());
166  itk::ImageRegionIteratorWithIndex<ImageType> sit(simg, img->GetLargestPossibleRegion());
167  sit.GoToBegin();
168  oit.GoToBegin();
169 
170  itk::Array<double> params(3);
171  params.Fill(0.0);
172  double count = 0.0;
173  itk::Point<double, 3> point;
174  for (; !oit.IsAtEnd(); ++oit, ++sit) {
175  if (oit.Get() != this->background_) {
176  sit.Set(oit.Get());
177  // Get the physical index from the image index.
178  img->TransformIndexToPhysicalPoint(oit.GetIndex(), point);
179  for (unsigned int i = 0; i < 3; i++) {
180  params[i] += point[i];
181  }
182  count += 1.0;
183  } else { sit.Set(this->background_); }
184  }
185 
186  // Compute center of mass.
187  for (unsigned int i = 0; i < 3; i++) {
188  params[i] = params[i] / count;
189  }
190 
191  double new_origin[3];
192  new_origin[0] = -(img->GetLargestPossibleRegion().GetSize()[0] / 2.0) * img->GetSpacing()[0];
193  new_origin[1] = -(img->GetLargestPossibleRegion().GetSize()[1] / 2.0) * img->GetSpacing()[1];
194  new_origin[2] = -(img->GetLargestPossibleRegion().GetSize()[2] / 2.0) * img->GetSpacing()[2];
195 
196  img->SetOrigin(new_origin);
197  if (this->verbose_) {
198  std::cerr << "new origin = " << img->GetOrigin() << "\n";
199  }
200  // Zero out the original image.
201  for (oit.GoToBegin(); !oit.IsAtEnd(); ++oit){
202  oit.Set(this->background_);
203  }
204 
205  // Translate the segmentation back into the original image.
206  itk::TranslationTransform<double, 3>::Pointer trans
207  = itk::TranslationTransform<double, 3>::New();
208  trans->SetParameters(params);
209 
210  itk::NearestNeighborInterpolateImageFunction<ImageType, double>::Pointer
211  interp = itk::NearestNeighborInterpolateImageFunction<ImageType, double>::New();
212 
213  itk::ResampleImageFilter<ImageType, ImageType>::Pointer resampler
214  = itk::ResampleImageFilter<ImageType, ImageType>::New();
215  resampler->SetOutputParametersFromImage(img);
216  resampler->SetTransform(trans);
217  resampler->SetInterpolator(interp);
218  resampler->SetInput(simg);
219  resampler->Update();
220 
221  oit.GoToBegin();
222  itk::ImageRegionIterator<ImageType> it(resampler->GetOutput(), img->GetLargestPossibleRegion());
223  for (; !it.IsAtEnd(); ++it, ++oit) {
224  oit.Set(it.Get());
225  }
226  this->images_[i] = resampler->GetOutput();
227  }
228  if (this->verbose_) {
229  std::cout << "*** FINISHED RUNNING TOOL: center" << std::endl;

234  if (this->verbose_) {
235  std::cout << "*** RUNNING TOOL: auto_pad on " <<
236  (which == -1 ? "all" : std::to_string(which)) << std::endl;
237  }
238  bool first = true;
239  auto start = (which == -1 ? 0 : which);

241  if (!this->paddingInit_) {
242  for (size_t i = 0; i < this->images_.size(); i++) {
243  auto img = this->images_[i];
244  if (first == true) {
245  first = false;
246  this->lower_ = img->GetLargestPossibleRegion().GetIndex();
247  this->upper_ = lower_ + img->GetLargestPossibleRegion().GetSize();
248  } else {
249  // Keep the largest bounding box.
250  ImageType::RegionType::IndexType lowerTmp
251  = img->GetLargestPossibleRegion().GetIndex();
252  ImageType::RegionType::IndexType upperTmp
253  = lowerTmp + img->GetLargestPossibleRegion().GetSize();
254  for (unsigned int i = 0; i < 3; i++) {
255  if (lowerTmp[i] < this->lower_[i]) {
256  this->lower_[i] = lowerTmp[i];
257  }
258  if (upperTmp[i] > this->upper_[i]){
259  this->upper_[i] = upperTmp[i];
260  }
261  }
262  }
263  }
264  this->paddingInit_ = true;
265  }
266  if (this->verbose_) {
267  std::cout << "Lower bound = " << this->lower_[0] << " " << this->lower_[1]
268  << " " << this->lower_[2] << std::endl;
269  std::cout << "Upper bound = " << this->upper_[0] << " " << this->upper_[1] << " "
270  << this->upper_[2] << std::endl;
271  }
272  // Make sure the origin is at the center of the image.
273  double orig[3];
274  for (unsigned int i = 0; i < 3; i++) {
275  orig[i] = -static_cast<double>(this->upper_[i] - this->lower_[i]) / 2.0;
276  }
277  for (size_t i = start; i < end; i++) {
278  auto img = this->images_[i];
279  itk::ConstantPadImageFilter<ImageType, ImageType>::Pointer padder
280  = itk::ConstantPadImageFilter<ImageType, ImageType>::New();
281  padder->SetConstant(0);
282  padder->SetInput(img);
283 
284  // set the desired padding
285  auto pd = static_cast<unsigned long>((this->padding_ + 1) / 2);
286  unsigned long hipad[3] = { pd, pd, pd };
287  unsigned long lowpad[3] = { pd, pd, pd };
288 
289  padder->SetPadUpperBound(hipad);
290  padder->SetPadLowerBound(lowpad);
291  padder->UpdateLargestPossibleRegion();
292 
293  if (this->verbose_) {
294  std::cout << "input region = "
295  << img->GetBufferedRegion().GetSize()
296  << std::endl;
297  std::cout << "lowpad: " << lowpad[0] << " " << lowpad[1] << " " << lowpad[2]
298  << std::endl;
299  std::cout << "hipad: " << hipad[0] << " " << hipad[1] << " " << hipad[2]
300  << std::endl;
301  }
302  this->images_[i] = padder->GetOutput();
303  }
304  if (this->verbose_) {
305  std::cout << "*** FINISHED RUNNING TOOL: auto_pad" << std::endl;
306  }
