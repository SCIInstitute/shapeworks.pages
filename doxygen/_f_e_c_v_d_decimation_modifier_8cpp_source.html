32 FEMesh* FECVDDecimationModifier::Apply(FEMesh* pm)
33 {
34  // make sure this is a triangle mesh
35  if (pm->IsType(FE_TRI3) == false)
36  {
37  return 0;
38  }
39 
40  // do the initialization
41  if (Initialize(pm) == false) return 0;
42 
43  // Minimize energy
44  if (Minimize(pm) == false) return 0;
45 
46  // create the new mesh
47  // we can create either the clustered mesh or the final decimated mesh
48  FEMesh* pnew = 0;
49  if (m_bcvd)
50  {
51  // partition mesh based on cluster assingments
52  pnew = CalculateCVD(pm);
53  }
54  else
55  {
56  // triangulate the decimation
57  pnew = Triangulate2(pm);
58 
59  // make sure we have a mesh
60  if (pnew == 0) return 0;
61 
62  // next, we use the auto-mesher to reconstruct all faces, edges and nodes
63  FEAutoMesher mesher;
64  mesher.BuildMesh(pnew);
65 
66  //interpolating shell thickness
67  InterpolateShellThickness(pnew);
68 
69  // TODO: I noticed that this algorithm can create duplicate elements. I'm not sure yet if
70  // this is a bug or if there is something fishy about this algorithm. In any case,
71  // for now I will explicitly check for duplicate elements and remove duplicates. Even
72  // this may not fix all problems, since sometimes duplicate elements can have different
73  // winding.
74  // remove duplicate elements
75  pnew->RemoveDuplicateElements();
76  mesher.BuildMesh(pnew);
77 
78  // TODO: Another problem I found is that some elements become non-manifold, meaning that
79  // one or two edges don't have any neighbors. If the origianl mesh is closed, then
80  // the decimated mesh must be closed as well. (Off course, this assumes that the original
81  // mesh is closed. Perhaps I can find another criteria for non-manifold).
82  pnew->RemoveNonManifoldElements();
83  mesher.BuildMesh(pnew);
84 
85  // TODO: Some elements can be wound in the wrong direction. So, we try to find those elements
86  // and invert them.
87  pnew->FixElementWinding();
88  mesher.BuildMesh(pnew);
89 
90  // TODO: These operations can create holes in the mesh. therefore we fill all holes.
91  FEFillHole fill;
92  fill.FillAllHoles(pnew);
93  }
94 
95  // All done!
96  return pnew;
97 }

105 bool FECVDDecimationModifier::Initialize(FEMesh* pm)
106 {
107  // nodes on original mesh
108  int N0 = pm->Nodes();
109 
110  // target number of clusters/vertices
111  int NC = (int) (m_pct*N0);
112  if (NC <= 0) return false;
113 
114  // allocate cluster data
115  // number of clusters equals number of target nodes
116  // plus one, the "null" cluster
117  m_Cluster.resize(NC + 1);
118 
119  // assign all triangles to the "null" cluster
120  int T0 = pm->Faces();
121  m_tag.assign(T0, 0);
122 
123  // getting the selected elements
124  vector<int> selectedFaces;
125  for(int i = 0 ; i < pm->Elements() ; i++)
126  {
127  if (pm->Element(i).IsSelected()) selectedFaces.push_back(i);
128  }
129 
130  // assign NC random triangles to a cluster
131  // TODO: make a better algorithm
132  int MAX_TRIES = 2*T0;
133  int ntries = 0;
134  int nc = 0;
135  if (selectedFaces.empty() == false)
136  {
137  // % of total cluster are from selected regions
138  int nc_sel = (int) (m_sel_pct * NC);
139 
140  // make sure there are fewer selected elements than nc_sel
141  int sel_faces = selectedFaces.size();
142  if (sel_faces < nc_sel) return false;
143 
144  // assign the first nc_sel seeds
145  while (nc < nc_sel)
146  {
147  int n = randi(sel_faces);
148  int face_num = selectedFaces[n];
149  if (m_tag[face_num] == 0) m_tag[face_num] = ++nc;
150 
151  ntries++;
152  if (ntries > MAX_TRIES) return false;
153  }
154  }
155 
156  // assign the remaining seeds
157  while (nc < NC)
158  {
159  int n = randi(T0);
160  if (m_tag[n] == 0) m_tag[n] = ++nc;
161 
162  ntries++;
163  if (ntries > MAX_TRIES) return false;
164  }
165 
166  // calculate the "rho" value (here, the area) and gamma (centroid) for each face
167  m_rho.resize(T0);
168  m_gamma.resize(T0);
169  vec3d r[3];
170  //Creating a node node list
171  FENodeNodeList NNL(pm);
172  NNL.Build();
173  for (int i=0; i<T0; ++i)
174  {
175  FEFace& fi = pm->Face(i);
176 
177  double curvature1 = 0;
178  double curvature2 = 0;
179  for(int j = 0; j<3; j++)
180  {
181  //normal vector to the jth node of the face
182  //step -1
183  vec3d p = fi.m_nn[j];
184  vec3d r0 = pm->Node(fi.n[j]).r;
185 
186  //step -2
187  vec3d r1 = vec3d(1.f- p.x*p.x , -p.y*p.x , -p.x*p.z);
188  r1.Normalize();
189  vec3d r3 = p;
190  vec3d r2 = r3 ^ r1;
191  mat3d R;
192  R[0][0] = r1.x; R[0][1] = r1.y; R[0][2] = r1.z;
193  R[1][0] = r2.x; R[1][1] = r2.y; R[1][2] = r2.z;
194  R[2][0] = r3.x; R[2][1] = r3.y; R[2][2] = r3.z;
195 
196  //step -3
197  int current_node = fi.n[j];
198  vector<vec3d> x;
199  x.reserve(NNL.Valence(current_node));
200  for (int k = 0; k<NNL.Valence(current_node);k++)
201  x.push_back(pm->Node(NNL.Node(current_node, k)).r);
202  vector<vec3d> y;
203  y.resize(x.size());
204  int nn = x.size();
205 
206  //step 4
207  // map coordinates
208  for(int k=0; k < nn; ++k)
209  {
210  vec3d tmp = x[k] - r0;
211  y[k] = R*tmp;
212  }
213 
214  //step 5
215  //Prepare linear system of equations.
216  matrix RR(nn, 3);
217  vector<double> r(nn);
218  for (int k=0; k<nn; ++k)
219  {
220  vec3d& p = y[k];
221  RR[k][0] = p.x*p.x;
222  RR[k][1] = p.x*p.y;
223  RR[k][2] = p.y*p.y;
224  r[k] = p.z;
225  }
226 
227  // solve for quadric coefficients
228  vector<double> q(3);
229  RR.lsq_solve(q, r);
230  double a = q[0], b = q[1], c = q[2];
231 
232  //step 6
233  //get the curvature
234  double k1 = a + c + sqrt((a-c)*(a-c) + b*b);
235  double k2 = a + c - sqrt((a-c)*(a-c) + b*b);
236  curvature1 += k1;
237  curvature2 += k2;
238  }
239  curvature1 /= 3.0;
240  curvature2 /= 3.0;
241 
242  // get the nodal coordinates
243  r[0] = pm->Node(fi.n[0]).r;
244  r[1] = pm->Node(fi.n[1]).r;
245  r[2] = pm->Node(fi.n[2]).r;
246 
247  // calculate rho and gamma
248  m_rho[i] = area_triangle(r) * pow(sqrt(curvature1 * curvature1 + curvature2 * curvature2), m_gradient);
249 
250  // make sure rho is positive
251  if (m_rho[i] <= 0.0) return false;
252 
253  // assign the centroid
254  m_gamma[i] = (r[0] + r[1] + r[2])/3.0;
255  }
256 
257  // now, calculate the sgamma and srho for each cluster
258  for (int i=0; i<T0; ++i)
259  {
260  int nc = m_tag[i];
261  if (nc > 0)
262  {
263  m_Cluster[nc].m_srho += m_rho[i];
264  m_Cluster[nc].m_sgamma += m_gamma[i]*m_rho[i];
265  }
266  }
267 
268  // build the edge list
269  m_Edge.clear();
270  for (int i=0; i<T0; ++i)
271  {
272  FEFace& fi = pm->Face(i);
273  for (int j=0; j<3; ++j)
274  {
275  int nj = fi.m_nbr[j];
276 
277  // this algorithm currently only works with closed surfaces
278  // so all neighbors must be assigned
279  if (nj < 0) return false;
280 
281  if (m_tag[i] < m_tag[nj])
282  {
283  EDGE e;
284  e.face[0] = i;
285  e.face[1] = nj;
286  if (i == nj) return false;
287 
288  e.node[0] = fi.n[j];
289  e.node[1] = fi.n[(j+1)%3];
290  m_Edge.push_back(e);
291  }
292  }
293  }
294 
295  return true;
296 }

328 bool FECVDDecimationModifier::Minimize(FEMesh* pm)
329 {
330  bool bconv = false;
331  const int MAX_ITER = 50000;
332  int niter = 0;
333  do
334  {
335  // let's be optimistic
336  bconv = true;
337 
338  // loop over all edges
339  list<EDGE>::iterator it;
340  for (it = m_Edge.begin(); it != m_Edge.end();)
341  {
342  EDGE& e = *it;
343 
344  // faces
345  int m = e.face[0];
346  int n = e.face[1];
347 
348  // clusters
349  int k = m_tag[m];
350  int l = m_tag[n];
351 
352  if (k == l)
353  {
354  it = m_Edge.erase(it);
355  bconv = false;
356  }
357  else
358  {
359  Cluster& Ck = m_Cluster[k];
360  Cluster& Cl = m_Cluster[l];
361 
362  // case 1 - no change
363  double L1 = (Ck.m_sgamma*Ck.m_sgamma)/Ck.m_srho + (Cl.m_sgamma*Cl.m_sgamma)/Cl.m_srho;
364 
365  // case 2 - face m to Cl
366  vec3d g0k = Ck.m_sgamma - m_gamma[m]*m_rho[m];
367  double r0k = Ck.m_srho - m_rho[m];
368 
369  vec3d g0l = Cl.m_sgamma + m_gamma[m]*m_rho[m];
370  double r0l = Cl.m_srho + m_rho[m];
371  double L2 = (g0k*g0k) / r0k + (g0l*g0l) / r0l;
372 
373  // case 3 - face n to Ck
374  vec3d g1k = Ck.m_sgamma + m_gamma[n]*m_rho[n];
375  double r1k = Ck.m_srho + m_rho[n];
376 
377  vec3d g1l = Cl.m_sgamma - m_gamma[n]*m_rho[n];
378  double r1l = Cl.m_srho - m_rho[n];
379  double L3 = (g1k*g1k)/r1k + (g1l*g1l)/r1l;
380 
381  if ((k==0) || ((L2>L1)&&(L2>L3)))
382  {
383  // case 2 wins
384  Ck.m_srho = r0k;
385  Ck.m_sgamma = g0k;
386  Cl.m_srho = r0l;
387  Cl.m_sgamma = g0l;
388 
389  if (Swap(pm->Face(m), m, l) == false) return false;
390 
391  bconv = false;
392  }
393  else if ((l==0) || ((L3>L1)&&(L3>L2)))
394  {
395  // case 3 wins
396  Ck.m_srho = r1k;
397  Ck.m_sgamma = g1k;
398  Cl.m_srho = r1l;
399  Cl.m_sgamma = g1l;
400 
401  if (Swap(pm->Face(n), n, k) == false)
402  return false;
403 
404  bconv = false;
405  }
406  else
407  {
408  // case 1 wins
409  // no change
410  ++it;
411  }
412  }
413  }
414 
415  if (bconv)
416  {
417  // In some cases, (really bad meshes) a few elements will not have
418  // been assigned to a cluster. If this happens, we just return an
419  // error and the user needs to clean up the mesh before decimating
420  int NF = pm->Faces();
421  for (int i=0; i<pm->Faces(); ++i)
422  {
423  if (m_tag[i] == 0)
424  return false;
425  }
426 
427  // If a face becomes isolated, (i.e. surrounded by different clusters)
428  // we reassign it to the "null" cluster
429  /* for (int i=0; i<pm->Faces(); ++i)
430  {
431  FEFace& f = pm->Face(i);
432  int m = m_tag[i];
433 
434  if (m_Cluster[m].faces() > 1)
435  {
436  int nl = 0;
437  for (int j=0; j<3; ++j)
438  {
439  int nj = f.m_nbr[j];
440  if (nj >= 0)
441  {
442  int mj = m_tag[nj];
443  if (m == mj) nl++;
444  }
445  }
446  if (nl == 0)
447  {
448  m_tag[i] = 0;
449  //m_Cluster[m].m_nf--;
450  bconv = false;
451  }
452  }
453  }
454  */
455  }
456 
457  niter++;
458  }
459  while ((bconv == false)&&(niter < MAX_ITER));
460 
461  if (niter >= MAX_ITER) return false;
462 
463  return true;
464 }

710 FEMesh* FECVDDecimationModifier::Triangulate2(FEMesh* pm)
711 {
712  // let's build the node data
713  int N = pm->Nodes();

715  for (int i=0; i<N; ++i)

721  FENodeFaceList NFL(pm);

730  return 0;
731  }
732  }
733 
734  // build the cluster's fid array
735  for (int i=0; i<pm->Faces(); ++i)
736  {
737  Cluster& C = m_Cluster[m_tag[i]];
738  C.m_fid.push_back(i);
739  }
740 
741  // number of nodes for decimanted mesh equals number of clusters
742  int nodes = (int) m_Cluster.size() - 1;
743 
744  // create a new mesh
745  FEMesh* pnew = new FEMesh;
746  pnew->Create(nodes, 0);
747  m_st.resize(nodes);
748  // calculate the node positions
749  for (int i=0; i<nodes; ++i)
750  {
751  FENode& nd = pnew->Node(i);
752  Cluster& Ci = m_Cluster[i+1];
753 
754  //nd.r = Ci.m_sgamma / Ci.m_srho;//change here //original point
755  vec3d po = Ci.m_sgamma / Ci.m_srho;
756  vec3d po_1 = vec3d(0,0,0); //projection of po on one of the faces
757  //find projection of this center back to original surface of the mesh
758  int flag = 0;
759  double thickness = 0;
760  for (int j =0 ;j<Ci.faces() ;j++)
761  {
762  FEFace& fj = pm->Face(Ci.m_fid[j]);
763  vec3d p[3];
764  p[0] = pm->Node(fj.n[0]).r;//p1
765  p[1] = pm->Node(fj.n[1]).r;//p2
766  p[2] = pm->Node(fj.n[2]).r;//p3
767 
768  vec3d u = p[1] - p[0]; //p2 - p1
769  vec3d v = p[2] - p[0]; //p3 - p1
770  vec3d n = u ^ v;//u x v
771  vec3d w = po - p[0]; //po - p1
772  double n_2 = n * n; //n^2
773  vec3d u_X_w = u ^ w;//u x w
774  double gama = (u_X_w * n)/n_2;
775  vec3d w_X_v = w ^ v;//w x v
776  double beta = (w_X_v * n)/n_2;
777  double alpha = 1 - gama - beta;
778  if (alpha <= 1 && alpha >= 0 && beta <= 1 && beta >= 0 && gama <= 1 && gama >= 0 )
779  {
780  po_1 = (p[0] * alpha) + (p[1] * beta) + (p[2] * gama);
781  flag = 1;
782  break;
783  }
784  }
785  if (flag == 0)
786  nd.r = po;
787  else
788  nd.r = po_1;
789 
790  //shell thickness
791  for (int j = 0; j<Ci.faces(); j++)
792  {
793  //shell thickness.....or some other variable

795  FEElement* ele = pm->ElementPtr(fi.m_elem[0]);
796  double *st = ele->m_h;

798  for(int ii = 0; ii< ele->Nodes();ii++)
799  temp_thickness += st[ii];
800  thickness += temp_thickness/ele->Nodes();
801  }
802  m_st[i] = thickness / Ci.faces();
803  }
804 
805  // list of triangles
806  FEFillHole fill;
807  vector<FEFillHole::FACE> tri_list;
808  tri_list.reserve(nodes);
809 
810  // loop over all nodes
811  for (int i=0; i<N; ++i)
812  {
813  NODE& nd = Node[i];
814  if (nd.nc == 3)
815  {
816  FEFillHole::FACE f;
817  f.n[0] = nd.c[0]-1; assert(nd.c[0] > 0);
818  f.n[1] = nd.c[1]-1; assert(nd.c[1] > 0);

827

829  fill.DivideRing(ring, tri);
830 
831  // append to the tri_list
832  tri_list.insert(tri_list.end(), tri.begin(), tri.end());
833  }
834  }
835

839
