250 void Visualizer::update_lut()
251 {
252  int num_points = this->cached_mean_.size() / 3;
253 
254  this->glyph_lut_->SetNumberOfTableValues( num_points + 1 );
255  this->glyph_lut_->SetTableRange( 0.0, (double)num_points + 1.0 );
256 
257  if ( this->selected_point_one_ < 0 )
258  {
259  this->glyph_lut_->ForceBuild();
260  }
261  else
262  {
263 
264  if ( this->selected_point_one_ >= 0 && this->selected_point_two_ >= 0 )
265  {
266  // measurement mode
267 
268  for ( int i = 0; i < num_points; i++ )
269  {
270  this->glyph_lut_->SetTableValue( i, 0, 0, 0 );
271 
272  if ( this->selected_point_one_ == i )
273  {
274  this->glyph_lut_->SetTableValue( i, 1, 1, 1 );
275  }
276 
277  if ( this->selected_point_two_ == i )
278  {
279  this->glyph_lut_->SetTableValue( i, 1, 1, 0 );
280  }
281  }
282  }
283  else
284  {
285  // color by distance from the selected point
286 
287  double p1[3];
288  p1[0] = this->cached_mean_[ this->selected_point_one_ * 3 + 0];
289  p1[1] = this->cached_mean_[ this->selected_point_one_ * 3 + 1];
290  p1[2] = this->cached_mean_[ this->selected_point_one_ * 3 + 2];
291 
292  std::vector<double> distances;
293  double max_distance = 0;
294  for ( int i = 0; i < num_points; i++ )
295  {
296 
297  double p2[3];
298  p2[0] = this->cached_mean_[ i * 3 + 0];
299  p2[1] = this->cached_mean_[ i * 3 + 1];
300  p2[2] = this->cached_mean_[ i * 3 + 2];
301 
302  double distance = sqrt( vtkMath::Distance2BetweenPoints( p1, p2 ) );
303  distances.push_back( distance );
304  if ( distance > max_distance )
305  {
306  max_distance = distance;
307  }
308  }
309 
310  vtkSmartPointer<vtkLookupTable> lut = vtkSmartPointer<vtkLookupTable>::New();
311  lut->SetTableRange( 0, max_distance );
312  lut->SetHueRange( 0, 1 );
313  lut->SetSaturationRange( 1, 1 );
314  lut->SetValueRange( 1, 1 );
315  lut->Build();
316 
317  for ( int i = 0; i < num_points; i++ )
318  {
319 
320  unsigned char* color = lut->MapValue( max_distance - distances[i] );
321 
322  //this->glyph_lut_->SetTableValue( i, distances[i] / max_distance, 0, 1 );
323  this->glyph_lut_->SetTableValue( i, color[0] / 255.0f, color[1] / 255.0f, color[2] / 255.0f );
324 
325  if ( this->selected_point_one_ == i )
326  {
327  this->glyph_lut_->SetTableValue( i, 1, 1, 1 );
328  }
329 
330  if ( this->selected_point_two_ == i )
331  {
332  this->glyph_lut_->SetTableValue( i, 0, 0, 0 );
333  }
334  }
335  }
336  }
337 
338  this->glyph_lut_->Modified();
339  this->lightbox_->set_glyph_lut( this->glyph_lut_ );
340 }
