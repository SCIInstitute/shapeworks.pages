146 void OptimizeTool::loadCutPlanesFile(std::string file) {
147  std::ifstream planes(file.c_str());
148  if (!planes.good()) {
149  emit error_message("Cannot open file: " + file);
150  this->ui_->cutPlanesFile->setText("None Selected");
151  this->cutPlanes_.clear();
152  return;
153  }
154  std::vector<std::array<itk::Point<float>, 3 > > cutPlanes;
155  while (planes.good()) {
156  float x1, y1, z1, x2, y2, z2, x3, y3, z3;
157  planes >> x1 >> y1 >> z1 >> x2 >> y2 >> z2 >> x3 >> y3 >> z3;
158  if (!planes.fail()) {
159  itk::Point<float> p1, p2, p3;
160  p1[0] = x1; p1[1] = y1; p1[2] = z1;
161  p2[0] = x2; p2[1] = y2; p2[2] = z2;
162  p3[0] = x3; p3[1] = y3; p3[2] = z3;
163  cutPlanes.push_back({ { p1, p2, p3 } });
164  }
165  }
166  planes.close();
167  if (cutPlanes.empty() || (cutPlanes.size() > 1 &&
168  cutPlanes.size() != this->project_->get_shapes().size())) {
169  emit error_message(std::string("Error reading cutting plane file. Must have 1 ") +
170  "set of 3 points, or X set of 3 points, where X is number of samples.");
171  this->ui_->cutPlanesFile->setText("None Selected");
172  this->cutPlanes_.clear();
173  return;
174  }
175  emit message("Successfully read in cut planes file: " + file);
176  this->cutPlanes_ = cutPlanes;
177 
178 }

191 void OptimizeTool::on_reconstructionButton_clicked() {
192  this->update_preferences();
193  emit message("Please wait: running reconstruction step...");
194  emit progress(5);
195  this->ui_->run_optimize_button->setEnabled(false);
196  this->ui_->reconstructionButton->setEnabled(false);
197  QThread *thread = new QThread;
198  std::vector<std::vector<itk::Point<float> > > local, global;
199  std::vector<ImageType::Pointer> images;
200  auto shapes = this->project_->get_shapes();
201  local.resize(shapes.size());
202  global.resize(shapes.size());
203  images.resize(shapes.size());
204  size_t ii = 0;
205  for (auto &s : shapes) {
206  auto l = s->get_local_correspondence_points();
207  auto g = s->get_global_correspondence_points();
208  for (size_t i = 0; i < l.size(); i+=3) {
209  itk::Point<float> pt, pt2;
210  pt[0] = l[i];
211  pt[1] = l[i + 1];
212  pt[2] = l[i + 2];
213  pt2[0] = g[i];
214  pt2[1] = g[i + 1];
215  pt2[2] = g[i + 2];
216  local[ii].push_back(pt);
217  global[ii].push_back(pt2);
218  }
219  images[ii] = s->get_groomed_image();
220  ii++;
221  }
222  ShapeworksWorker *worker = new ShapeworksWorker(
223  ShapeworksWorker::Reconstruct, NULL, NULL, this->project_,
224  local, global, images,
225  this->ui_->maxAngle->value(),
226  this->ui_->meshDecimation->value(),
227  this->ui_->numClusters->value());
228  worker->moveToThread(thread);
229  connect(thread, SIGNAL(started()), worker, SLOT(process()));
230  connect(worker, SIGNAL(result_ready()), this, SLOT(handle_reconstruction_complete()));
231  connect(worker, SIGNAL(error_message(std::string)), this, SLOT(handle_error(std::string)));
232  connect(worker, SIGNAL(warning_message(std::string)), this, SLOT(handle_warning(std::string)));
233  connect(worker, SIGNAL(message(std::string)), this, SLOT(handle_message(std::string)));
234  connect(worker, SIGNAL(finished()), worker, SLOT(deleteLater()));
235  thread->start();
236  emit progress(15);
237 }
