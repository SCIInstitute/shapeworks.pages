132  FENodeNodeList NNL(pnew);
133  NNL.Build();
134  vector<vec3d>phi_node;
135  phi_node.reserve(pnew->Nodes());
136  for(int j =0 ;j<m_iteration;j++)
137  {
138  for(int i = 0; i < pnew->Nodes() ; i++)
139  {
140  FENode& ni = pnew->Node(i);

142  for (int k = 0; k<NNL.Valence(i);k++)
143  {
144  vec3d x = pnew->Node(NNL.Node(i, k)).r;
145  r_sum += x;
146  }
147  r_sum = r_sum/NNL.Valence(i);
148  r_sum -= ni.r;
149  phi_node.push_back(r_sum);

151  for(int i = 0; i < pnew->Nodes() ; i++)
152  {
153  if(hashmap[i] == 0)
154  {
155  FENode& ni = pnew->Node(i);
156  vec3d phi_old = phi_node[i];
157 
158  vec3d r_sq_sum,phi_sq_old;
159  for (int k = 0; k<NNL.Valence(i);k++)
160  {
161  int neigh_node = NNL.Node(i, k);
162  r_sq_sum += phi_node[neigh_node];
163  }
164  phi_sq_old = r_sq_sum/NNL.Valence(i);
165  phi_sq_old -= phi_old;
166 
167  ni.r = ni.r - (phi_old * (m_threshold2 - m_threshold1)) - (phi_sq_old *(m_threshold1*m_threshold2));
168  }
169  }

174 void FEMeshSmoothingModifier::Crease_Enhancing_Diffusion(FEMesh* pnew,vector<int> hashmap)
175 {
176  //creating Node Element list
177  FENodeFaceList NFL(pnew);
178  NFL.Build();
179 
180  //calculating m(R) for each face i.e for each triangle
181  vector<FEFace*> mR ;
182  vector<vec3d>m_R;
183  vector<vec3d>m_R_new;
184  mR.reserve(pnew->Faces());
185  m_R.reserve(pnew->Faces());
186  m_R_new.reserve(pnew->Faces());
187 
188  //for first iteration m_R are normals
189  for(int i =0; i< pnew->Faces();i++)
190  {
191  FEFace& fa = pnew->Face(i);
192  m_R.push_back(fa.m_fn);
193  }
194  for (int iter = 0 ; iter< m_iteration;iter++)
195  {
196  //for each face calculate m_R
197  for(int i =0;i<pnew->Faces();i++)
198  {
199  FEFace& fa = pnew->Face(i);
200  vec3d centroid_R = (pnew->Node(fa.n[0]).r + pnew->Node(fa.n[1]).r + pnew->Node(fa.n[2]).r )/3;
201  //finding the neighbouring faces
202  for(int j =0 ;j<3 ;j++)
203  {
204  int nodeID = fa.n[j];
205  for (int k = 0; k<NFL.Valence(nodeID);k++)
206  {
207  FEFace *fa1 = NFL.Face(nodeID,k);
208  if(fa1->m_ntag != 1 && fa1->m_elem[0] != i)
209  {
210  fa1->m_ntag = 1;
211  mR.push_back(fa1);
212  }
213  }
214  }
215  //now we have the neighbouring faces in mR list
216  double weight =0;
217  m_R_new.push_back(vec3d(0,0,0));
218  for(int k =0;k<mR.size();k++)
219  {
220  FEFace *fa1 = mR[k];
221  vec3d r[3]; //three nodes of the face
222  r[0] = pnew->Node(fa1->n[0]).r;
223  r[1] = pnew->Node(fa1->n[1]).r;
224  r[2] = pnew->Node(fa1->n[2]).r;
225  vec3d centroid_S = (r[0]+r[1]+r[2])/3;
226  double dist = distance(centroid_S,centroid_R);
227  double angle = acos((fa.m_fn * fa1->m_fn)/(fa.m_fn.Length() * fa1->m_fn.Length()));//angle between the normals
228  double weight1 = area_triangle(r) * exp(-m_threshold1 * angle*angle*dist*dist);
229  weight += weight1;
230  m_R_new[i] += m_R[fa1->m_elem[0]] * weight1;
231  fa1->m_ntag = 0;
232  }
233  m_R_new[i] = m_R_new[i]/weight;
234  mR.clear();
235  }
236  //we have m_R_new for each face.
237  for(int i =0 ;i<m_R_new.size();i++)
238  {
239  m_R[i] = m_R_new[i];
240  }
241  m_R_new.clear();
242  //For each node modify its coodinates
243  for(int i = 0 ;i < pnew->Nodes();i++)
244  {
245  if(hashmap[i] == 0) //not the edge node
246  {
247  FENode& ni = pnew->Node(i);
248  vec3d vR;
249  double weight=0;
250  for (int k = 0; k<NFL.Valence(i);k++)
251  {
252  FEFace *fa1 = NFL.Face(i,k);
253  vec3d r[3]; //three nodes of the face
254  r[0] = pnew->Node(fa1->n[0]).r;
255  r[1] = pnew->Node(fa1->n[1]).r;
256  r[2] = pnew->Node(fa1->n[2]).r;
257  vec3d centroid = (r[0]+r[1]+r[2])/3;
258  weight += area_triangle(r);
259  vec3d PC = centroid - ni.r;
260  double temp = PC * m_R[fa1->m_elem[0]];
261  vR += (m_R[fa1->m_elem[0]] * temp)*area_triangle(r);
262  }
263  vR = vR/weight;
264  ni.r = ni.r + vR;
265  }
266  }
267  }//end of one iteration
268 }
